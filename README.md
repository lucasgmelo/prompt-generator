### PROMPT-API

## DEPENDÊNCIAS 
Python >= 3.9
Node.js

## SETUP
# Criar Python Virtual env no diretório generators

cd generators
python -m  venv venv

.\venv\Scripts\activate

# Instalar depedências do Python
pip install -r requirements.txt

# Configurando o servidor

cd api
npm install yarn
npm install

## RUN
# SERVER
cd api
yarn start
